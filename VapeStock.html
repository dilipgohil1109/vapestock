Google Drive Sync

How it works: Stock data is saved as a JSON file in your Google Drive. Both phones read and write to the same file — any change syncs within seconds.

One Google account. Both phones use the same Web App URL.

Step 1 — One-time Setup
Do this once. Creates the shared file in Google Drive.
1
Go to script.google.com → New Project
2
Delete existing code, paste the code below, then click Save
3
Deploy → New Deployment → Web App → Execute as: Me → Anyone → Deploy
4
Authorize → copy the Web App URL → paste below
Apps Script Code
function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var file = getFile();
  file.setContent(JSON.stringify(data));
  return ContentService.createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  var file = getFile();
  var content = file.getContent();
  return ContentService.createTextOutput(content || '{}')
    .setMimeType(ContentService.MimeType.JSON);
}

function getFile() {
  var folder = DriveApp.getRootFolder();
  var files = folder.getFilesByName('vapestock_data.json');
  if (files.hasNext()) return files.next();
  return folder.createFile('vapestock_data.json', '{}', MimeType.PLAIN_TEXT);
}
